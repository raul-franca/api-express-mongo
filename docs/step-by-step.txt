
package.json :
    add: "type": "module", // para ativar a importação/exportação do módulo ES6

criar o arquivo servidor.js

npm install nodemon
add no package.json:
    "scripts": {
        "start": "nodemon servidor.js"
    }

npm install express

cria src/app.js

altera servidor.js
para:
import app from "./src/app.js";

const PORT = 3000;

app.listen(PORT, () => {
    console.log(`Server is running on port ${PORT}`);
});
